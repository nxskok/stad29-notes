data prefs;
  infile "mlogit.dat" delimiter=",";
  input brand sex age;

  class brand;
  class sex;
  model brand=sex age / link=glogit;
  output out=mlogit2 pred=pred lower=lcl upper=ucl;

proc print data=mlogit2;

run;
